% Shortcuts to create glossary entries with (nested) subscripts and superscripts in various forms (e.g., bold, roman, with bar or dot)
% The macros check if the subscript/superscript exists as a glossary entry and use it if so, otherwise the literal text is used.
%
% The macros require that the first (main) symbol is a glossary entry.
%
% For symbols using numerical or explicit subscripts, use \gls{label}[_123]
%
% Examples:
% \gsub{a}{i} produces a_i
% \gsubrm{a}{i} produces a_{\mathrm{i}}
% \gbarsub{a}{i} produces \bar{a}_i
% \gbarsubrm{a}{i} produces \bar{a}_{\mathrm{i}}
% \gbarsup{a}{i} produces \bar{a}^i
% \gbarsuprm{a}{i} produces \bar{a}^{\mathrm{i}}
% \gdotsub{a}{i} produces \dot{a}_i
% \gdotsubrm{a}{i} produces \dot{a}_{\mathrm{i}}
% \gboldsub{a}{i} produces \mathbf{a}_i
% \gsubbold{a}{i} produces a_{\mathbf{i}}
% \gboldsubbold{a}{i} produces \mathbf{a}_{\mathbf{i}}
% \gboldsubrm{a}{i} produces \mathbf{a}_{\mathrm{i}}


% Symbols with subscripts

\newcommand*{\gsub}[2]{%
	\ensuremath{%
		\gls{#1}[_{%
			\ifglsentryexists{#2}{%
				\gls{#2}}%
			{#2}%
		}]%
	}%
}%

\newcommand*{\gsubrm}[2]{%
	\ensuremath{%
		\gls{#1}[_{%
			\ifglsentryexists{#2}{%
				\mathrm{\gls{#2}}}%
			{\mathrm{#2}}%
		}]%
	}%
}%
